-- =====================================================
-- 1. LOGIN SERVICE DATABASE
-- =====================================================
USE login_db;

CREATE TABLE IF NOT EXISTS users (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    github_id BIGINT NULL,
    github_access_token VARCHAR(255) NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_username (username),
    INDEX idx_email (email)
);

CREATE TABLE IF NOT EXISTS sessions (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id BIGINT NOT NULL,
    jwt_token TEXT NOT NULL,
    expires_at TIMESTAMP NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    INDEX idx_user_id (user_id),
    INDEX idx_expires_at (expires_at)
);

-- =====================================================
-- 2. PROJECT SERVICE DATABASE
-- =====================================================
USE project_db;

CREATE TABLE IF NOT EXISTS projects (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    owner_id BIGINT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_owner_id (owner_id),
    INDEX idx_name (name)
);

CREATE TABLE IF NOT EXISTS repositories (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    project_id BIGINT NOT NULL,
    github_repo_id BIGINT NOT NULL,
    repo_name VARCHAR(100) NOT NULL,
    repo_full_name VARCHAR(200) NOT NULL,
    repo_url VARCHAR(500) NOT NULL,
    clone_url VARCHAR(500),
    default_branch VARCHAR(50) DEFAULT 'main',
    is_active BOOLEAN DEFAULT TRUE,
    last_scan_at TIMESTAMP NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (project_id) REFERENCES projects(id) ON DELETE CASCADE,
    UNIQUE KEY uk_project_repo (project_id, github_repo_id),
    INDEX idx_project_id (project_id),
    INDEX idx_repo_full_name (repo_full_name),
    INDEX idx_last_scan_at (last_scan_at)
);

-- =====================================================
-- 3. SCAN SERVICE DATABASE (FIXED - NO ENUMS)
-- =====================================================
USE scan_db;

-- Scans table
CREATE TABLE IF NOT EXISTS scans (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    project_id BIGINT NOT NULL,
    repository_id BIGINT NOT NULL,
    repository_url VARCHAR(500) NOT NULL,
    branch VARCHAR(50) DEFAULT 'main',
    scan_status VARCHAR(20) DEFAULT 'PENDING',
    sonarqube_project_key VARCHAR(200),
    started_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP NULL,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_project_id (project_id),
    INDEX idx_repository_id (repository_id),
    INDEX idx_scan_status (scan_status),
    INDEX idx_started_at (started_at),
    INDEX idx_created_at (created_at)
);

-- Vulnerabilities table (FIXED: VARCHAR instead of ENUM)
CREATE TABLE IF NOT EXISTS vulnerabilities (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    scan_id BIGINT NOT NULL,
    sonarqube_rule_id VARCHAR(100),
    vulnerability_type VARCHAR(50),
    severity VARCHAR(20) DEFAULT 'MAJOR',
    file_path VARCHAR(500),
    line_number INT,
    message TEXT,
    status VARCHAR(20) DEFAULT 'OPEN',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (scan_id) REFERENCES scans(id) ON DELETE CASCADE,
    INDEX idx_scan_id (scan_id),
    INDEX idx_severity (severity),
    INDEX idx_status (status),
    INDEX idx_file_path (file_path(255))
);

-- AI Suggestions table (FIXED: DOUBLE for confidence_score)
CREATE TABLE IF NOT EXISTS ai_suggestions (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    vulnerability_id BIGINT NOT NULL UNIQUE,
    suggestion_text TEXT NOT NULL,
    code_example TEXT,
    confidence_score DOUBLE DEFAULT 0.85,
    model_used VARCHAR(50) DEFAULT 'deepseek-chat',
    generated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (vulnerability_id) REFERENCES vulnerabilities(id) ON DELETE CASCADE,
    INDEX idx_vulnerability_id (vulnerability_id),
    INDEX idx_generated_at (generated_at)
);

-- Scan summary table
CREATE TABLE IF NOT EXISTS scan_summary (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    scan_id BIGINT NOT NULL UNIQUE,
    total_vulnerabilities INT DEFAULT 0,
    blocker_count INT DEFAULT 0,
    critical_count INT DEFAULT 0,
    major_count INT DEFAULT 0,
    minor_count INT DEFAULT 0,
    info_count INT DEFAULT 0,
    summary_json JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (scan_id) REFERENCES scans(id) ON DELETE CASCADE,
    INDEX idx_scan_id (scan_id)
);
